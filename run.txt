docker build  -t bigjenkins -f dockerfile .  


docker run -d \
  --name jenkins \
  -u root \
  -p 8090:8080 \
  -p 50000:50000 \
  --restart unless-stopped \
  --privileged \
  -v my-jenkins-data:/var/jenkins_home \
  -v /var/run/docker.sock:/var/run/docker.sock \
  -v $HOME/.kube:/root/.kube \
  bigjenkins
